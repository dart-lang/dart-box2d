ClipVertex class

Used for computing contact manifolds.

class ClipVertex {
 Vector2 v;
 ContactID id;

 ClipVertex() :
   v = new Vector2.zero(),
   id = new ContactID() { }

 void setFrom(ClipVertex cv){
   v.setFrom(cv.v);
   id.setFrom(cv.id);
 }
}
