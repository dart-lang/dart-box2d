SimplexVertex class 

class SimplexVertex {
 final vec2 wA; // support point in shapeA
 final vec2 wB; // support point in shapeB
 final vec2 w; // wB - wA
 num a; // barycentric coordinate for closest point
 int indexA; // wA index
 int indexB; // wB index

 SimplexVertex() :
   wA = new vec2.zero(),
   wB = new vec2.zero(),
   w = new vec2.zero(),
   a = 0,
   indexA = 0,
   indexB = 0;

 void setFrom(SimplexVertex sv) {
   wA.setFrom(sv.wA);
   wB.setFrom(sv.wB);
   w.setFrom(sv.w);
   a = sv.a;
   indexA = sv.indexA;
   indexB = sv.indexB;
 }

 String toString() => "wA: $wA, wB: $wB, w: $w";
}
